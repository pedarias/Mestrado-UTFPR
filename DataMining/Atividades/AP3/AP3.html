# Funo atualizada de limpeza de dados
def Limpeza_Dados(Texto):
    # Remoo de links
    etapa_01 = re.sub('www\S+|http\S+', '', Texto)
    # Transformar para minsculo
    etapa_02 = etapa_01.lower()
    # Remoo de caracteres especiais
    etapa_03 = re.sub(r'[!~@#$%^&*()+=|{}[\]:;<.>?/\'\\",-]', ' ', etapa_02)
    # Remoo de nmeros
    etapa_04 = re.sub('[0-9]', '', etapa_03)
    # Remove espaos extras
    etapa_05 = re.sub(r'\s+', ' ', etapa_04)
    return etapa_05

# Aplicar a limpeza nos dados
df['Review_cleaned'] = df['Review'].apply(Limpeza_Dados)

# Avaliao por review positivo e negativo

review_positive = ''
review_negative = ''

# Verifying each class (positive and negative)
for c in df['class'].unique():
    # Filter the dataframe for the current class
    filter_df = df[df['class'] == c]
    
    # Iterate over the filtered dataframe
    for text in filter_df['text']:
        # If the class is 'Positive Rating', concatenate to review_positive
        if c == 'Positive Rating':
            review_positive += ' ' + ' '.join(text.split())
        # If the class is 'Negative Rating', concatenate to review_negative
        elif c == 'Negative Rating':
            review_negative += ' ' + ' '.join(text.split())

# Trim the leading spaces
review_positive = review_positive.strip()
review_negative = review_negative.strip()

In[]:
def Remover_StopWords(Texto):

  Lista_Palavras = Texto.split()

  # Texto sem as stopwords
  nova_frase = ''

  for word in Lista_Palavras:
    if word not in lista_stopwords:
      nova_frase = nova_frase + ' ' + word

  return nova_frase

def Extrair_Radical(Texto):

  Lista_Palavras = Texto.split()
  nova_frase = ''

  for word in Lista_Palavras:
    radical = Stem.stem(word)
    nova_frase = nova_frase + ' ' + radical
  return nova_frase
